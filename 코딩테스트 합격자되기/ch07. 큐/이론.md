# 큐의 동작 원리
![](https://prmoreira23.github.io/assets/queue-data-structure.gif)
- 연산
  - isFull(): 큐에 들어있는 데이터의 수가 최대인지 확인하는 연산
  - isEmpty(): 큐에 데이터가 하나도 없는지 확인하는 연산
  - enqueue(): 큐에 데이터를 삽입하는 연산
  - dequeue(): 큐에서 데이터를 삭제하는 연산
- 변수/상태
  - peek(): 큐에서 가장 먼저 들어온 데이터를 가리키는 변수
  - front: 큐에서 가장 먼저 들어온 데이터를 가리키는 변수
  - rear: 큐에서 가장 나중에 들어온 데이터를 가리키는 변수
  - data[maxSize]: 큐의 데이터를 저장하는 배열

# 큐의 특성을 활용하는 분야
- 작업 대기열: 네트워크 통신을 할 때 다수의 클라이언트에서 서버에 작업 요청하면, 서버는 요청이 들어온 순서대로 작업을 처리
- 이벤트 처리: 어떤 어플리케이션이나 시스템에서 사용자의 이벤트(예: 키보드 입력)를 처리할 때 사용

# 큐 구현
## Shift() 메서드 사용하기
```js
const queue = [];

// enqueue
queue.push(1);
queue.push(2);
queue.push(3);

// 큐의 맨 앞 데이터 제거
let firstItem = queue.shift();
console.log(firstItem); // 1

// 큐에 데이터 추가
queue.push(4);
queue.push(5);

// 큐의 맨 앞 데이터 제거
firstItem = queue.shift();
console.log(firstItem); // 2
```

## 배열을 이용하는 방식
```js
class Queue {
    constructor() {
        this.items = [];
        this.front = 0; // 큐의 맨 앞 인덱스
        this.rear = 0;  // 큐의 맨 뒤 인덱스
    }
    
    // 큐에 데이터 추가
    enqueue(item) {
        this.items.push(item);
        this.rear++;
    }
    
    // 큐에서 데이터 제거
    dequeue() {
        return this.items[this.front++];
    }
    
    // 큐가 비어있는지 확인
    isEmpty() {
        return this.front === this.rear;
    }
}
```

## 연결리스트를 이용하는 방식
```js
class Node {
    constructor(data) {
        this.data = data; // 노드의 데이터
        this.next = null; // 다음 노드를 가리키는 포인
    }
}

class Queue {
    constructor() {
        this.head = null; // 큐의 맨 앞 노드
        this.tail = null; // 큐의 맨 뒤 노드
        this.size = 0;   // 큐의 크기
    }
    
    // 큐에 데이터 추가
    enqueue(data) {
        const newNode = new Node(data);
        
        if (!this.head) {
            this.head = newNode; // 큐가 비어있으면 head와 tail을 새 노드로 설정
            this.tail = newNode;
        } else {
            this.tail.next = newNode; // 큐가 비어있지 않으면 tail의 next를 새 노드로 설정
            this.tail = newNode;      // tail을 새 노드로 업데이트
        }
        
        this.size++; // 큐의 크기 증가
    }
    
    // 큐에서 데이터 제거
    dequeue() {
        if (!this.head) return null; // 큐가 비어있으면 null 반환
        
        // 두 번째 요소를 head의 참조로 변경하면
        // 자연스럽게 첫 번째 요소가 사라짐
        const removeNode = this.head;
        this.head = this.head.next;
        
        // 만약 두 번째 요소가 없다면
        // 큐가 비어 있다는 뜻이니 tail도 null로 설정
        if (!this.head) {
            this.tail = null;
        }
        
        this.size--; // 큐의 크기 감소
        
        return removeNode.data; // 제거된 노드의 데이터 반환
    }
    
    isEmpty() {
        return this.size === 0; // 큐가 비어있는지 확인
    }
}
```
